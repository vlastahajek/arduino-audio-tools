210 void processClient() {
211 //LOGD("processClient");
212 if (client_obj) { // if you get a client,
213 LOGI("New Client."); // print a message out the serial port
214 String currentLine = ""; // make a String to hold incoming data from the client
215 while (client_obj.connected()) { // loop while the client's connected
216 if (client_obj.available()) { // if there's bytes to read from the client,
217 char c = client_obj.read(); // read a byte, then
218 if (c == '\n') { // if the byte is a newline character
219
220 // if the current line is blank, you got two newline characters in a row.
221 // that's the end of the client HTTP request, so send a response:
222 if (currentLine.length() == 0){
223 sendReplyHeader();
224 sendReplyContent();
225 // break out of the while loop:
226 break;
227 } else { // if you got a newline, then clear currentLine:
228 currentLine = "";
229 }
230 }
231 else if (c != '\r') { // if you got anything else but a carriage return character,
232 currentLine += c; // add it to the end of the currentLine
233 }
234 }
