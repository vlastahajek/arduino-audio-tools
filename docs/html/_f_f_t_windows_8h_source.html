23class WindowFunction {
24 public:
25 WindowFunction() = default;
26
27 virtual void begin(int samples) {
28 this->samples_minus_1 = -1.0f + samples;
29 this->i_samples = samples;
30 }
31
32 inline float ratio(int idx) {
33 return (static_cast<float>(idx) - 1.0 / samples_minus_1);
34 }
35
36 inline int samples() { return i_samples; }
37 virtual float factor(int idx) = 0;

39 protected:

53class BufferedWindow : public WindowFunction {
54 public:
55 BufferedWindow(WindowFunction* wf) { p_wf = wf; }
56 BufferedWindow(BufferedWindow const&) = delete;
57 BufferedWindow& operator=(BufferedWindow const&) = delete;
58
59
60 virtual void begin(int samples) {
61 // process only if there is a change
62 if (p_wf->samples() != samples) {
63 p_wf->begin(samples);
64 len = samples / 2;
65 if (p_buffer != nullptr) delete[] p_buffer;
66 p_buffer = new float[len];
67 for (int j = 0; j < len; j++) {
68 p_buffer[j] = p_wf->factor(j);
69 }
70 }
71 }
72
73 ~BufferedWindow() {
74 if (p_buffer != nullptr) delete[] p_buffer;
75 }
76
77 inline float factor(int idx) {
78 return idx < len ? p_buffer[idx] : p_buffer[i_samples - idx];
79 }
80
81 protected:
82 WindowFunction* p_wf = nullptr;
83 float* p_buffer = nullptr;
84 int len;
85};
