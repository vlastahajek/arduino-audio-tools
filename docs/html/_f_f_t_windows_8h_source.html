23class WindowFunction {
24 public:
25 WindowFunction() = default;
26
27 virtual void begin(int samples) {
28 this->samples_minus_1 = -1.0f + samples;
29 this->i_samples = samples;
30 }
31
32 inline float ratio(int idx) {
33 return (static_cast<float>(idx) - 1.0 / samples_minus_1);
34 }
35
36 inline int samples() { return i_samples; }
37 virtual float factor(int idx) = 0;
38
39 protected:
40 float samples_minus_1 = 0;

53class BufferedWindow : public WindowFunction {
54 public:
55 BufferedWindow(WindowFunction* wf) { p_wf = wf; }
56
57 virtual void begin(int samples) {
58 // process only if there is a change
59 if (p_wf->samples() != samples) {
60 p_wf->begin(samples);
61 len = samples / 2;
62 if (p_buffer != nullptr) delete[] p_buffer;
63 p_buffer = new float[len];
64 for (int j = 0; j < len; j++) {
65 p_buffer[j] = p_wf->factor(j);
66 }
67 }
68 }
69
70 ~BufferedWindow() {
71 if (p_buffer != nullptr) delete[] p_buffer;
72 }
73
74 inline float factor(int idx) {
75 return idx < len ? p_buffer[idx] : p_buffer[i_samples - idx];
76 }
77
78 protected:
79 WindowFunction* p_wf = nullptr;
80 float* p_buffer = nullptr;
81 int len;
82};
