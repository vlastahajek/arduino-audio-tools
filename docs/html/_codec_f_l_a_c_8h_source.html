159 static FLAC__StreamDecoderReadStatus read_callback(const FLAC__StreamDecoder *decoder, FLAC__byte result_buffer[],size_t *bytes, void *client_data) {
160 FLAC__StreamDecoderReadStatus result = FLAC__STREAM_DECODER_READ_STATUS_CONTINUE;
161 LOGI("read_callback: %d", *bytes);
162 FLACDecoder *self = (FLACDecoder *)client_data;
163 if (self == nullptr || !self->is_active) {
164 return FLAC__STREAM_DECODER_READ_STATUS_ABORT;
165 }
166
167 // get data directly from stream
168 *bytes = self->readBytes(result_buffer, *bytes);
169 LOGD("-> %d", *bytes);
170 if (self->isEof(*bytes)){
171 result = FLAC__STREAM_DECODER_READ_STATUS_END_OF_STREAM;
172 self->is_active = false;
173 }
174 return result;
175 }

178 bool isEof(int bytes) {
179 bool result = false;
180 if (bytes==0){
181 delay(5);
182 } else {
183 time_last_read=millis();
184 }
185 if (millis() - time_last_read >= read_timeout_ms){
186 result = true;
187 }
188 return result;

192 static FLAC__StreamDecoderWriteStatus write_callback(const FLAC__StreamDecoder *decoder, const FLAC__Frame *frame,const FLAC__int32 *const buffer[], void *client_data) {
193 LOGI("write_callback: %d", frame->header.blocksize);
194 FLACDecoder *self = (FLACDecoder *)client_data;
195
196 AudioBaseInfo actual_info = self->audioInfo();
197 if (self->info != actual_info){
198 self->info = actual_info;
199 self->info.logInfo();
200 int bps = FLAC__stream_decoder_get_bits_per_sample(decoder);
201 if (bps!=16){
202 LOGI("Converting from %d bits", bps);
203 }
204 if (self->p_notify != nullptr) {
205 self->info = actual_info;
206 self->p_notify->setAudioInfo(self->info);
207 }
208 }
209
210 // write audio data
211 int bps = FLAC__stream_decoder_get_bits_per_sample(decoder);
212 int16_t sample;
213 switch(bps){
214 case 8:
215 for (int j = 0; j < frame->header.blocksize; j++) {
216 for (int i = 0; i < actual_info.channels; i++) {
217 //self->output_buffer[j*actual_info.channels + i] = buffer[i][j]<<8;
218 sample = buffer[i][j]<<8;;
219 self->p_print->write((uint8_t *)&sample,2);
220 }
221 }
222 break;
223 case 16:
224 for (int j = 0; j < frame->header.blocksize; j++) {
225 for (int i = 0; i < actual_info.channels; i++) {
226 //self->output_buffer[j*actual_info.channels + i] = buffer[i][j];
227 sample = buffer[i][j];
228 self->p_print->write((uint8_t *)&sample,2);
229 }
230 }
231 break;
232 case 24:
233 for (int j = 0; j < frame->header.blocksize; j++) {
234 for (int i = 0; i < actual_info.channels; i++) {
235 //self->output_buffer[j*actual_info.channels + i] = buffer[i][j]>>8;
236 sample = buffer[i][j] >>8;
237 self->p_print->write((uint8_t *)&sample,2);
238 }
239 }
240 break;
241 case 32:
242 for (int j = 0; j < frame->header.blocksize; j++) {
243 for (int i = 0; i < actual_info.channels; i++) {
244 //self->output_buffer[j*actual_info.channels+ i] = buffer[i][j]>>16;
245 sample = buffer[i][j] >>16;
246 self->p_print->write((uint8_t *)&sample,2);
247 }
248 }
249 break;
250 default:
251 LOGE("Unsupported bps: %d", bps);
252 }
253
254 return FLAC__STREAM_DECODER_WRITE_STATUS_CONTINUE;
255 }

357 virtual size_t write(const void *in_ptr, size_t in_size) override {
358 if (!is_open || p_print == nullptr) return 0;
359 LOGD("write: %u", in_size);
360 size_t result = 0;
361 int samples=0;
362 int frames=0;
363 int32_t *data=nullptr;
364 switch(cfg.bits_per_sample){
365 case 16:
366 samples = in_size / sizeof(int16_t);
367 frames = samples / cfg.channels;
368 writeBuffer((int16_t*)in_ptr, samples);
369 data = buffer.data();
370 break;
371
372 case 32:
373 samples = in_size / sizeof(int32_t);
374 frames = samples / cfg.channels;
375 data = (int32_t*) in_ptr;
376 break;
377
378 default:
379 LOGE("bits_per_sample not supported: %d", cfg.bits_per_sample);
380 break;
381 }
382
383 if (frames>0){
384 if (FLAC__stream_encoder_process_interleaved(p_encoder, data, frames)){
385 result = in_size;

387 LOGE("FLAC__stream_encoder_process_interleaved");
