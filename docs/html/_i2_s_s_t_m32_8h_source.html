111 size_t readBytes(void *dest, size_t size_bytes){
112 TRACED();
113 if (cfg.channels == 2){
114 return p_rx_buffer->readArray((uint8_t*)dest, size_bytes);
115 } else {
116 // combine two channels to 1: so request double the amout
117 int req_bytes = size_bytes*2;
118 uint8_t tmp[req_bytes];
119 int16_t *tmp_16 = (int16_t*) tmp;
120 int eff_bytes = p_rx_buffer->readArray((uint8_t*)tmp, req_bytes);
121 // add 2 channels together
122 int16_t *dest_16 = (int16_t *)dest;
123 int16_t eff_samples = eff_bytes / 2;
124 int16_t idx = 0;
125 for (int j=0;j<eff_samples;j+=2){
126 dest_16[idx++] = static_cast<float>(tmp_16[j])+tmp_16[j+1] / 2.0;
127 }
128 return eff_bytes / 2;
129 }

227 size_t writeBytesExt(const void *src, size_t size_bytes){
228 size_t result = 0;
229 if (cfg.channels == 2){
230 result = p_tx_buffer->writeArray((uint8_t*)src, size_bytes);
231 } else {
232 // write each sample 2 times
233 int samples = size_bytes / 2;
234 int16_t *src_16 = (int16_t *)src;
235 int16_t tmp[2];
236 int result = 0;
237 for (int j=0;j<samples;j++){
238 tmp[0]=src_16[j];
239 tmp[1]=src_16[j];
240 if (p_tx_buffer->availableForWrite()>=4){
241 p_tx_buffer->writeArray((uint8_t*)tmp, 4);
242 result = j*2;
243 } else {
244 // abort if the buffer is full
245 break;
246 }
247 }
248 }
249 LOGD("writeBytesExt: %u", result)
250 return result;
