100 bool begin(VS1053Config cfg) {
101 TRACEI();
102 bool result = true;
103 // enfornce encoded data for midi mode
104 if (cfg.is_midi){
105 cfg.is_encoded_data = true;
106 }
107 this->cfg = cfg;
108 setAudioInfo(cfg);
109 cfg.logInfo();
110 LOGI("is_encoded_data: %s", cfg.is_encoded_data?"true":"false");
111 LOGI("is_midi: %s", cfg.is_midi?"true":"false");
112 LOGI("cs_pin: %d", cfg.cs_pin);
113 LOGI("dcs_pin: %d", cfg.dcs_pin);
114 LOGI("dreq_pin: %d", cfg.dreq_pin);
115 LOGI("reset_pin: %d", cfg.reset_pin);
116 LOGI("cs_sd_pin: %d", cfg.cs_sd_pin);
117
118 if (p_vs1053==nullptr){
119 p_vs1053 = new VS1053(cfg.cs_pin,cfg.dcs_pin,cfg.dreq_pin);
120 p_vs1053_out = new VS1053StreamOut(p_vs1053);
121
122 if (cfg.is_start_spi) {
123 LOGI("SPI.begin()")
124 SPI.begin();
125 } else {
126 LOGI("SPI not started");
127 }
128
129 if (cfg.reset_pin!=-1){
130 LOGI("Setting reset pin to high: %d", cfg.reset_pin);
131 pinMode(cfg.reset_pin, OUTPUT);
132 digitalWrite(cfg.reset_pin, HIGH);
133 delay(800);
134 }
135 }
136
137 // Output Stream
138 if (p_out==nullptr){
139 AudioEncoder *p_enc = cfg.is_encoded_data ? &copy:p_encoder;
140 p_out = new EncodedAudioStream(p_vs1053_out, p_enc);
141 }
142
143 // hack to treat midi as separate mode
144 const int MIDI_MODE = 100;
145 int mode = cfg.mode;
146 if (cfg.is_midi){
147 mode = MIDI_MODE;
148 }
149
150 switch(mode){
151
152 case TX_MODE:
153 result = beginTx();
154 break;
155#if VS1053_EXT
156 case MIDI_MODE:
157 result = beginMidi();
158 break;
159
160 case RX_MODE:
161 result = beginRx();
162 break;
163#endif
164 default:
165 LOGD("Mode not supported");
166 result = false;
167 break;
168 }
169 return result;
170 }
