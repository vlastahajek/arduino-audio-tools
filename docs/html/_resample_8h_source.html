278 size_t upsampleMultiply(T *from, T* to, int sample_count, int channels, int factor ){
279 int frame_count = sample_count/channels;
280 memset(to,0, sample_count*factor*2);
281 // just repeat each frame facor times
282 int idx = 0;
283 for (int j=0;j<frame_count;j++){
284 for (int f=0;f<factor;f++){
285 for (int ch=0;ch<channels;ch++){
286 to[idx++] = from[j+ch];
287 }
288 }
289 }

295 size_t upsample(T *from, T* to, int sample_count, int channels, int factor ){
296 int frame_count = sample_count/channels;
297 size_t result = 0;
298 int pos = 0;
299 // we start with the last saved frame
300 for (int16_t frame_pos=-1; frame_pos<frame_count-1; frame_pos++){
301 for (int8_t ch=0; ch<channels; ch++){
302 T actual_data = *p_data(frame_pos, ch, from);
303 T next_data = *p_data(frame_pos+1, ch, from);
304 float diff = (next_data - actual_data) / factor;
305 pos =(frame_pos+1)*factor;
306 *p_data(pos, ch, to) = actual_data;
307 result++;
308 for (int16_t f=1;f<=factor;f++){
309 pos = ((frame_pos+1)*factor)+f;
310 T tmp = actual_data + (diff*f);
311 *p_data(pos, ch, to) = tmp;
312 result++;
313 }
314 }
315 }
316 // save last frame_pos data
317 for (int8_t ch=0; ch<channels; ch++){
318 T tmp = *p_data(frame_count-1, ch, from);
319 last_end[ch] = tmp;
320 }
321 return frame_count*factor*channels;
322 }
