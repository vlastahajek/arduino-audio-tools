275 virtual size_t write(const void *in_ptr, size_t in_size) override {
276 if (!is_open || p_print == nullptr) return 0;
277 LOGD("write: %d", (int) in_size);
278
279 if (p_codec!=nullptr){
280 // encode the data
281 size_t eff = p_encoded_audio_stream->write((uint8_t*)in_ptr, in_size);
282 if (eff!=in_size){
283 LOGE("Write overflow");
284 }
285 // get the result from the buffer
286 void *encoded_data = buffer.address();
287 int enoded_size = buffer.available();
288
289 op.packet = (uint8_t *)encoded_data;
290 op.bytes = enoded_size;
291 } else {
292 op.packet = (uint8_t *)in_ptr;
293 op.bytes = in_size;
294 }
295 if (op.bytes>0){
296 buffer.reset();
297 op.granulepos = granulepos +=
298 in_size / sizeof(int16_t) / cfg.channels; // sample
299 op.b_o_s = false;
300 op.e_o_s = false;

302 is_audio = true;
303 if (!writePacket(op, OGGZ_FLUSH_AFTER)) {
304 return 0;
305 }
306 }

309 ;
