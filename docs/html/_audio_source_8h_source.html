144 template<typename T, size_t N>
145 AudioSourceURL(AbstractURLStream& urlStream, T(&urlArray)[N], const char* mime, int startPos = 0) {
146 TRACED();
147 this->actual_stream = &urlStream;
148 this->mime = mime;
149 this->urlArray = urlArray;
150 this->max = N;
151 this->pos = startPos - 1;
152 this->timeout_auto_next_value = 20000;
153 }
154
156 virtual void begin() override {
157 TRACED();
158 this->pos = 0;
159 }
160
162 Stream* selectStream(int idx) override {
163 pos = idx;
164 if (pos < 0) {
165 pos = 0;
166 LOGI("url array out of limits: %d -> %d", idx, pos);
167 }
168 if (pos >= max) {
169 pos = max-1;
170 LOGI("url array out of limits: %d -> %d", idx, pos);
171 }
172 LOGI("selectStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
173 if (started) actual_stream->end();
174 actual_stream->begin(urlArray[pos], mime);
175 started = true;
176 return actual_stream;
177 }
178
180 Stream* nextStream(int offset) override {
181 pos += offset;
182 if (pos < 0 || pos >= max) {
183 pos = 0;
184 }
185 LOGI("nextStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
186 return selectStream(pos);
187 }
188
190 Stream* previousStream(int offset) override {
191 pos -= offset;
192 if (pos < 0 || pos >= max) {
193 pos = max-1;
194 }
195 LOGI("previousStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
196 return selectStream(pos);
197 }
198
200 Stream* selectStream(const char* path) override {
201 LOGI("selectStream: %s", path);
202 if (started) actual_stream->end();
203 actual_stream->begin(path, mime);
204 started = true;
205 return actual_stream;
206 }
207
208 int index() {
209 return pos;
210 }
211
212 const char *toStr() {
213 return urlArray[pos];
214 }
215
217 void setTimeout(int millisec){
218 actual_stream->setTimeout(millisec);
219 }
220
221 // provides go not to the next on error
222 virtual bool isAutoNext() {
223 return true;
224 };
225
226 // only the ICYStream supports this
227 bool setMetadataCallback(void (*fn)(MetaDataType info, const char* str, int len), ID3TypeSelection sel=SELECT_ICY) {
228 TRACEI();
229 return actual_stream->setMetadataCallback(fn);
230 }
231
232
233protected:
