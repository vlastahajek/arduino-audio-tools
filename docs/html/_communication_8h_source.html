194 size_t write(const uint8_t *data, size_t len) override {
195 int open = len;
196 size_t result = 0;
197 int retry_count = 0;
198 while (open > 0) {
199 if (available_to_write > 0) {
200 resetAvailableToWrite();
201 size_t send_len = min(open, ESP_NOW_MAX_DATA_LEN);
202 esp_err_t rc = esp_now_send(nullptr, data + result, send_len);
203 // wait for confirmation
204 if (cfg.use_send_ack) {
205 while (available_to_write == 0) {
206 delay(1);
207 }
208 } else {
209 is_write_ok = true;
210 }
211 // check status
212 if (rc == ESP_OK && is_write_ok) {
213 open -= send_len;
214 result += send_len;
215 } else {
216 LOGW("Write failed - retrying again");
217 retry_count++;
218 if (cfg.write_retry_count>0 && retry_count>=cfg.write_retry_count){
219 LOGE("Write error after %d retries", cfg.write_retry_count);
220 // break loop
221 return 0;
222 }
223 }
224 // if we do have no partner to write we stall and retry later
225 } else {
226 delay(cfg.delay_after_write_ms);
227 }
228
229 // Wait some time before we retry
230 if (!is_write_ok) {
231 delay(cfg.delay_after_failed_write_ms);
232 }
233 }
234 return result;
235 }

554 size_t write(const uint8_t *data, size_t len) override {
555 int written_len = 0;
556 int open_len = len;
557 AudioSendData send;
558 while (written_len < len) {
559 int available_to_write = waitForRequest();
560 size_t to_write_len = DEFAULT_BUFFER_SIZE;
561 to_write_len = min(open_len, available_to_write);
562 send.increment();
563 send.size = to_write_len;
564 p_dest->write((const uint8_t *)&send, sizeof(send));
565 int w = p_dest->write(data + written_len, to_write_len);
566 written_len += w;
567 open_len -= w;
568 }
569 return written_len;
570 }
571
572 int availableForWrite() override { return available_to_write; }
573
574 void end() {
575 AudioDataEnd end;
576 end.increment();
577 p_dest->write((const uint8_t *)&end, sizeof(end));
578 }

582 int available_to_write = 1024;
583 bool is_sync;

587 while (p_dest->available() < size) {

609class AudioSyncReader : public AudioStreamX {
610 public:
611 AudioSyncReader(Stream &in, EncodedAudioStream &out,
612 bool isConfirmer = true) {
613 p_in = &in;
614 p_out = &out;
615 is_confirmer = isConfirmer;
616 }
617
618 size_t copy() {
619 int processed = 0;
620 int header_size = sizeof(header);
621 waitFor(header_size);
622 readBytes((uint8_t *)&header, header_size);
623
624 switch (header.rec) {
625 case Begin:
626 audioDataBegin();
627 break;
628 case End:
629 audioDataEnd();
630 break;
631 case Send:
632 processed = receiveData();
633 break;
634 }
635 return processed;
636 }
637
638 protected:
639 Stream *p_in;
640 EncodedAudioStream *p_out;
641 AudioConfirmDataToReceive req;
642 AudioHeader header;
643 AudioDataBegin begin;
644 size_t available = 0; // initial value
645 bool is_started = false;
646 bool is_confirmer;
647 int last_seq = 0;
648
650 void audioDataBegin() {
651 readProtocol(&begin, sizeof(begin));
652 p_out->begin();
653 p_out->setAudioInfo(begin.info);
654 requestData();
655 }
656
658 void audioDataEnd() {
659 AudioDataEnd end;
660 readProtocol(&end, sizeof(end));
661 p_out->end();
662 }
663
664 // Receives audio data
665 int receiveData() {
666 AudioSendData data;
667 readProtocol(&data, sizeof(data));
668 available = data.size;
669 // receive and process audio data
670 waitFor(available);
671 int max_gap = 10;
672 if (data.seq > last_seq ||
673 (data.seq < max_gap && last_seq >= (32767 - max_gap))) {
674 uint8_t buffer[available];
675 p_in->readBytes((uint8_t *)buffer, available);
676 p_out->write((const uint8_t *)buffer, available);
677 // only one reader should be used as confirmer
678 if (is_confirmer) {
679 requestData();
680 }
681 last_seq = data.seq;
682 }
683 return available;
684 }
685
687 void waitFor(int size) {
688 while (p_in->available() < size) {
689 delay(10);
690 }
691 }
692
694 void requestData() {
695 req.size = p_out->availableForWrite();
696 req.increment();
697 p_in->write((const uint8_t *)&req, sizeof(req));
698 p_in->flush();
699 }
700
702 void readProtocol(AudioHeader *data, int len) {
703 const static int header_size = sizeof(header);
704 memcpy(data, &header, header_size);
705 int read_size = len - header_size;
706 waitFor(read_size);
707 readBytes((uint8_t *)data + header_size, read_size);
708 }
