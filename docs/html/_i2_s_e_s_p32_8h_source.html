95 size_t result = 0;
96 if (cfg.channels==2){

98 TRACEE();
99 }
100 } else if (cfg.channels==1){
101 // I2S has always 2 channels. We support to reduce it to 1
102 uint8_t temp[size_bytes*2];
103 if (i2s_read(i2s_num, temp, size_bytes*2, &result, portMAX_DELAY)!=ESP_OK){
104 TRACEE();

108 case 16: {

110 result = reducer16.convert((uint8_t*)dest,temp, result);

116 case 32: {
117 ChannelReducer<int32_t> reducer32(1, 2);
118 result = reducer32.convert((uint8_t*)dest, temp, result);
119 } break;
120 default:
121 LOGE("invalid bits_per_sample: %d", cfg.bits_per_sample);

148 .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,

209 static size_t writeExpandChannel(i2s_port_t i2s_num, const int bits_per_sample, const void *src, size_t size_bytes){
210 size_t result = 0;
211 int j;
212 switch(bits_per_sample){
213
214 case 8:
215 for (j=0;j<size_bytes;j++){
216 int8_t frame[2];
217 int8_t *data = (int8_t *)src;
218 frame[0]=data[j];
219 frame[1]=data[j];
220 size_t result_call = 0;
221 if (i2s_write(i2s_num, frame, sizeof(int8_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
222 TRACEE();
223 } else {
224 result += result_call;
225 }
226 }
227 break;
228
229 case 16:
230 for (j=0;j<size_bytes/2;j++){
231 int16_t frame[2];
232 int16_t *data = (int16_t*)src;
233 frame[0]=data[j];
234 frame[1]=data[j];
235 size_t result_call = 0;
236 if (i2s_write(i2s_num, frame, sizeof(int16_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
237 TRACEE();
238 } else {
239 result += result_call;
240 }
241 }
242 break;
243
244 case 24:
245 for (j=0;j<size_bytes/4;j++){
246 int24_t frame[2];
247 int24_t *data = (int24_t*) src;
248 frame[0]=data[j];
249 frame[1]=data[j];
250 size_t result_call = 0;
251 if (i2s_write(i2s_num, frame, sizeof(int24_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
252 TRACEE();
253 } else {
254 result += result_call;
255 }
256 }
257 break;
258
259 case 32:
260 for (j=0;j<size_bytes/4;j++){
261 int32_t frame[2];
262 int32_t *data = (int32_t*) src;
263 frame[0]=data[j];
264 frame[1]=data[j];
265 size_t result_call = 0;
266 if (i2s_write(i2s_num, frame, sizeof(int32_t)*2, &result_call, portMAX_DELAY)!=ESP_OK){
267 TRACEE();

269 result += result_call;
270 }
271 }
272 break;
273 }

327 i2s_mode_t mode;

342#endif
343 break;
344
345 default:
346 LOGW("signal_type undefined");
