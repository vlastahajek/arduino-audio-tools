77 audio_hal_iface_samples_t toSampleRate() {
78 TRACED();
79 const static int ia[] = {8000, 11025, 16000, 22050,
80 24000, 32000, 44100, 48000};
81 const static audio_hal_iface_samples_t oa[] = {
82 AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,
83 AUDIO_HAL_22K_SAMPLES, AUDIO_HAL_24K_SAMPLES, AUDIO_HAL_32K_SAMPLES,
84 AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
85 int diff = 99999;
86 int result = 0;
87 for (int j = 0; j < 8; j++) {
88 if (ia[j] == sample_rate) {
89 LOGD("-> %d",ia[j])
90 return oa[j];
91 } else {
92 int new_diff = abs(oa[j] - sample_rate);
93 if (new_diff < diff) {
94 result = j;
95 diff = new_diff;
96 }
97 }
98 }
99 LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
100 return oa[result];
101 }

239 virtual size_t readBytes(uint8_t *data, size_t length) override {
240 if (cfg.channels == 2) {
241 return kit.read(data, length);
242 } else if (cfg.channels==1) {
243 // convert 2 channels of int16_t to 1
244 int16_t temp[length];
245 int len_res = kit.read(temp, length*2);
246 int res_count = len_res / 2;
247 int16_t *out = (int16_t*) data;
248 for (int j=0; j<res_count; j+=2){
249 int32_t total = temp[j];
250 total+=temp[j+1];
251 total = total/2;
252 *out = total;
253 out++;
254 }
255 return res_count;
256 }
257 LOGE("Unsuported number of channels : %d", cfg.channels);
258 return 0;

263 virtual void setAudioInfo(AudioBaseInfo info) {
264 TRACEI();
265
266 if (cfg.sample_rate != info.sample_rate
267 && cfg.bits_per_sample == info.bits_per_sample
268 && cfg.channels == info.channels
269 && is_started) {
270 // update sample rate only
271 cfg.sample_rate = info.sample_rate;
272 cfg.logInfo();
273 converter.setAudioInfo(cfg);
274 kit.setSampleRate(cfg.toSampleRate());
275 } else if (cfg.sample_rate != info.sample_rate
276 || cfg.bits_per_sample != info.bits_per_sample
277 || cfg.channels != info.channels
278 || !is_started) {
279 // more has changed and we need to start the processing
280 cfg.sample_rate = info.sample_rate;
281 cfg.bits_per_sample = info.bits_per_sample;
282 cfg.channels = info.channels;
283 cfg.logInfo();
284
285 // Stop first
286 if(is_started){
287 kit.end();
288 }
289 // update input format
290 converter.setAudioInfo(cfg);
291 // start kit with new config
292 kit.begin(cfg.toAudioKitConfig());
293 is_started = true;
294 }

575 AudioActions::ActiveLogic getActionLogic(int pin){
576 input_key_service_info_t input_key_info[] = INPUT_KEY_DEFAULT_INFO();
577 int size = sizeof(input_key_info) / sizeof(input_key_info[0]);
578 for (int j=0; j<size; j++){
579 if (pin == input_key_info[j].act_id){
580 switch(input_key_info[j].type){
581 case PERIPH_ID_ADC_BTN:
582 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveHigh);
583 return AudioActions::ActiveHigh;
584 case PERIPH_ID_BUTTON:
585 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveLow);
586 return AudioActions::ActiveLow;
587 case PERIPH_ID_TOUCH:
588 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveTouch);
589 return AudioActions::ActiveTouch;
590 }
591 }
592 }
593 LOGW("Undefined ActionLogic for pin: %d ",pin);
594 return AudioActions::ActiveLow;

598 void setupActions() {
599 TRACEI();
600 // SPI might have been activated
601 if (!cfg.sd_active){
602 LOGW("Deactivating SPI because SD is not active");
603 SPI.end();
604 }
605
606 // pin conflicts with the SD CS pin for AIThinker and buttons
607 if (! (cfg.sd_active && (AUDIOKIT_BOARD==5 || AUDIOKIT_BOARD==6))){
608 LOGD("actionStartStop")
609 addAction(kit.pinInputMode(), actionStartStop);
610 } else {
611 LOGW("Mode Button ignored because of conflict: %d ",kit.pinInputMode());
612 }
613
614 // pin conflicts with AIThinker A101 and headphone detection
615 if (! (cfg.sd_active && AUDIOKIT_BOARD==6)) {
616 LOGD("actionHeadphoneDetection pin:%d",kit.pinHeadphoneDetect())
617 actions.add(kit.pinHeadphoneDetect(), actionHeadphoneDetection, AudioActions::ActiveChange);
618 } else {
619 LOGW("Headphone detection ignored because of conflict: %d ",kit.pinHeadphoneDetect());
620 }
621
622 // pin conflicts with SD Lyrat SD CS Pin and buttons / Conflict on Audiokit V. 2957
623 if (! (cfg.sd_active && (AUDIOKIT_BOARD==1 || AUDIOKIT_BOARD==7))){
624 LOGD("actionVolumeDown")
625 addAction(kit.pinVolumeDown(), actionVolumeDown);
626 LOGD("actionVolumeUp")
627 addAction(kit.pinVolumeUp(), actionVolumeUp);
628 } else {
629 LOGW("Volume Buttons ignored because of conflict: %d ",kit.pinVolumeDown());
630 }
631 }
