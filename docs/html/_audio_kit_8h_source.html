77 audio_hal_iface_samples_t toSampleRate() {
78 TRACED();
79 const static int ia[] = {8000, 11025, 16000, 22050,
80 24000, 32000, 44100, 48000};
81 const static audio_hal_iface_samples_t oa[] = {
82 AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,
83 AUDIO_HAL_22K_SAMPLES, AUDIO_HAL_24K_SAMPLES, AUDIO_HAL_32K_SAMPLES,
84 AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
85 int diff = 99999;
86 int result = 0;
87 for (int j = 0; j < 8; j++) {
88 if (ia[j] == sample_rate) {
89 LOGD("-> %d",ia[j])
90 return oa[j];
91 } else {
92 int new_diff = abs(oa[j] - sample_rate);
93 if (new_diff < diff) {
94 result = j;
95 diff = new_diff;
96 }
97 }
98 }
99 LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
100 return oa[result];
101 }

233 virtual size_t readBytes(uint8_t *data, size_t length) override {
234 if (cfg.channels == 2) {
235 return kit.read(data, length);
236 } else if (cfg.channels==1) {
237 // convert 2 channels of int16_t to 1
238 int16_t temp[length];
239 int len_res = kit.read(temp, length*2);
240 int res_count = len_res / 2;
241 int16_t *out = (int16_t*) data;
242 for (int j=0; j<res_count; j+=2){
243 int32_t total = temp[j];
244 total+=temp[j+1];
245 total = total/2;
246 *out = total;
247 out++;
248 }
249 return res_count;
250 }
251 LOGE("Unsuported number of channels : %d", cfg.channels);
252 return 0;

257 virtual void setAudioInfo(AudioBaseInfo info) {
258 TRACEI();
259
260 if (cfg.sample_rate != info.sample_rate
261 && cfg.bits_per_sample == info.bits_per_sample
262 && cfg.channels == info.channels
263 && is_started) {
264 // update sample rate only
265 cfg.sample_rate = info.sample_rate;
266 cfg.logInfo();
267 converter.setAudioInfo(cfg);
268 kit.setSampleRate(cfg.toSampleRate());
269 } else if (cfg.sample_rate != info.sample_rate
270 || cfg.bits_per_sample != info.bits_per_sample
271 || cfg.channels != info.channels
272 || !is_started) {
273 // more has changed and we need to start the processing
274 cfg.sample_rate = info.sample_rate;
275 cfg.bits_per_sample = info.bits_per_sample;
276 cfg.channels = info.channels;
277 cfg.logInfo();
278
279 // Stop first
280 if(is_started){
281 kit.end();
282 }
283 // update input format
284 converter.setAudioInfo(cfg);
285 // start kit with new config
286 kit.begin(cfg.toAudioKitConfig());
287 is_started = true;
288 }

569 AudioActions::ActiveLogic getActionLogic(int pin){
570 input_key_service_info_t input_key_info[] = INPUT_KEY_DEFAULT_INFO();
571 int size = sizeof(input_key_info) / sizeof(input_key_info[0]);
572 for (int j=0; j<size; j++){
573 if (pin == input_key_info[j].act_id){
574 switch(input_key_info[j].type){
575 case PERIPH_ID_ADC_BTN:
576 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveHigh);
577 return AudioActions::ActiveHigh;
578 case PERIPH_ID_BUTTON:
579 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveLow);
580 return AudioActions::ActiveLow;
581 case PERIPH_ID_TOUCH:
582 LOGD("getActionLogic for pin %d -> %d", pin, AudioActions::ActiveTouch);
583 return AudioActions::ActiveTouch;
584 }
585 }
586 }
587 LOGW("Undefined ActionLogic for pin: %d ",pin);
588 return AudioActions::ActiveLow;

592 void setupActions() {
593 TRACEI();
594 // SPI might have been activated
595 if (!cfg.sd_active){
596 LOGW("Deactivating SPI because SD is not active");
597 SPI.end();
598 }
599
600 // pin conflicts with the SD CS pin for AIThinker and buttons
601 if (! (cfg.sd_active && (AUDIOKIT_BOARD==5 || AUDIOKIT_BOARD==6))){
602 LOGD("actionStartStop")
603 addAction(kit.pinInputMode(), actionStartStop);
604 } else {
605 LOGW("Mode Button ignored because of conflict: %d ",kit.pinInputMode());
606 }
607
608 // pin conflicts with AIThinker A101 and headphone detection
609 if (! (cfg.sd_active && AUDIOKIT_BOARD==6)) {
610 LOGD("actionHeadphoneDetection pin:%d",kit.pinHeadphoneDetect())
611 actions.add(kit.pinHeadphoneDetect(), actionHeadphoneDetection, AudioActions::ActiveChange);
612 } else {
613 LOGW("Headphone detection ignored because of conflict: %d ",kit.pinHeadphoneDetect());
614 }
615
616 // pin conflicts with SD Lyrat SD CS Pin and buttons / Conflict on Audiokit V. 2957
617 if (! (cfg.sd_active && (AUDIOKIT_BOARD==1 || AUDIOKIT_BOARD==7))){
618 LOGD("actionVolumeDown")
619 addAction(kit.pinVolumeDown(), actionVolumeDown);
620 LOGD("actionVolumeUp")
621 addAction(kit.pinVolumeUp(), actionVolumeUp);
622 } else {
623 LOGW("Volume Buttons ignored because of conflict: %d ",kit.pinVolumeDown());
624 }
625 }
