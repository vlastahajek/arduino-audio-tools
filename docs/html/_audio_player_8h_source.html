112 virtual bool begin(int index=0, bool isActive = true) {
113 TRACED();
114 bool result = false;
115 // initilaize volume
116 if (current_volume==-1){
117 setVolume(1.0);
118 } else {
119 setVolume(current_volume);
120 }
121
122 // navigation support
123 autonext = p_source->isAutoNext();
124
125 // start dependent objects
126 p_out_decoding->begin();
127 p_source->begin();
128 meta_out.begin();
129
130 if (index >= 0) {
131 p_input_stream = p_source->selectStream(index);
132 if (p_input_stream != nullptr) {
133 if (meta_active) {
134 copier.setCallbackOnWrite(decodeMetaData, this);
135 }
136 copier.begin(*p_out_decoding, *p_input_stream);
137 timeout = millis() + p_source->timeoutAutoNext();
138 active = isActive;
139 result = true;
140 } else {
141 LOGW("-> begin: no data found");
142 active = false;
143 result = false;
144 }
145 } else {
146 LOGW("-> begin: no stream selected");
147 active = isActive;
148 result = false;
149 }
150 return result;
151 }

296 virtual void copy() {
297 if (active) {
298 TRACED();
299 if (p_final_print!=nullptr && p_final_print->availableForWrite()==0){
300 // not ready to do anything - so we wait a bit
301 delay(100);
302 return;
303 }
304 // handle sound
305 if (copier.copy() || timeout == 0) {
306 // reset timeout
307 timeout = millis() + p_source->timeoutAutoNext();
308 }
309 // move to next stream after timeout
310 if (p_input_stream == nullptr || millis() > timeout) {
311 if (autonext) {
312 if (previous_stream == false) {
313 LOGW("-> timeout - moving to next stream");
314 // open next stream
315 if (!next(1)) {
316 LOGD("stream is null");
317 }
318 } else {
319 LOGW("-> timeout - moving to previous stream");
320 // open previous stream
321 if (!previous(1)) {
322 LOGD("stream is null");
323 }
324 }
325 } else {
326 active = false;
327 }
328 timeout = millis() + p_source->timeoutAutoNext();
329 }
