258 // Add samples to input data p_x - and process them if full
259 template<typename T>
260 void processSamples(const void *data, size_t byteCount) {
261 T *dataT = (T*) data;
262 T sample;
263 float sample_windowed;
264 int samples = byteCount/sizeof(T);
265 for (int j=0; j<samples; j+=cfg.channels){
266 sample = dataT[j+cfg.channel_used];
267 sample_windowed = sample;
268 // optionally apply window function
269 if (cfg.window_function!=nullptr){
270 sample_windowed = cfg.window_function->factor(current_pos) * sample;
271 }
272 p_driver->setValue(current_pos, sample_windowed);
273 writeStrideBuffer((uint8_t*)&sample, sizeof(T));
274 if (++current_pos>=cfg.length){
275 fft();
276 }
277 }
278 }

280 void fft() {
281 timestamp_begin = millis();
282 p_driver->fft();
283 timestamp = millis();
284 if (cfg.callback!=nullptr){
285 cfg.callback(*this);
286 }
287
288 // reprocess data in stride buffer
289 if (p_stridebuffer!=nullptr){
290 // rewrite data in stride buffer
291 int byte_count = p_stridebuffer->available();
292 uint8_t buffer[byte_count];
293 p_stridebuffer->readArray(buffer, byte_count);
294 write(buffer, byte_count);
295 current_pos = byte_count / bytesPerSample();
296 } else {
297 current_pos = 0;
298 }

306 template<int N>
307 bool InsertSorted(AudioFFTResult(&result)[N], AudioFFTResult tmp){
308 for (int j=0;j<N;j++){
309 if (tmp.magnitude>result[j].magnitude){
310 // shift existing values right
311 for (int i=N-2;i>=j;i--){
312 result[i+1] = result[i];
313 }
314 // insert new value
315 result[j]=tmp;
316 }
317 }
318 return false;
319 }

321 void writeStrideBuffer(uint8_t* buffer, size_t len){
322 if (p_stridebuffer!=nullptr){
323 int available = p_stridebuffer->availableForWrite();
324 if (len>available){
325 // clear oldest values to make space
326 int diff = len-available;
327 for(int j=0;j<diff;j++){
328 p_stridebuffer->read();
329 }
330 }
331 p_stridebuffer->writeArray(buffer, len);
332 }
